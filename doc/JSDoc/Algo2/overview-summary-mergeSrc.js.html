var manager = new jsAnimManager(40); 

function Msort(bId,posx,posy)
{
    var topPosition = document.getElementById("box"+bId).offsetTop;
    topPosition += posy; 
    var leftPosition = document.getElementById("box"+bId).offsetLeft;
    leftPosition += posx;
    Move('box'+bId,leftPosition,topPosition, 1000);               
}

/* 
    Eases all of the Div Boxes, box0, .... ,box9
    red - 00 to 255
    green - 00 to 255
    blue - 00 to 255
    time - duration, 1 sec = 1000
*/
function EaseAll(red,green,blue,time) {
	for( i = 0; i < 10; i++ )
	{
		Ease('box'+i,red,green,blue,time);
	}
}

function Ease(id,red,green,blue,time) {
	 var anim = manager.createAnimObject(""+id+"");  
	 anim.add({
		property: Prop.backgroundColor, 
		to: new Col(red,green,blue),  
		duration: time}); 
}
function Kill() {
	anim.pause();
}
function Move(id,x,y,time) {
	var anim = manager.createAnimObject(""+id+"");  
   	anim.add({property: Prop.position, to: new Pos(x,y),   
     duration: time,
	 ease:	jsAnimEase.parabolicNeg  
	 });
}
function FallAnima(id,posx,posy,pos2y) {
	var anim = manager.createAnimObject(""+id+"");
	anim.add({
			  property:		Prop.position,
			  to:			new Pos(posx,posy),
			  duration:		300,
			  ease:			jsAnimEase.parabolicPos,
			  onComplete: 	function() {
				  				anim.add({
				  				property:		Prop.position,
								to:			    new Pos(posx,pos2y),
				    			duration:		600,
		    	     			ease:			jsAnimEase.bounceSmooth,
								} );
				  			}//Function Closed
			  });
}
function Transform(id,h,w,time) {
	var anim = manager.createAnimObject(""+id+"");  
   	anim.add({property: Prop.dimension, to: new Dim(h,w),   
     duration: time});
}
