var count = 5;
function Dijsktra() 
{
	var nShortPath;
	while(asPriority.length  > 1 )
	{
		nShortPath = asPriorityQ.pop();
		count -= 1;
		setTable(k);
		asShortPaths.push(nShortPath);
		if(nShortPath + asEdgeWeights[k] < asDistance[k+1])
		{
			asDistance[k+1] = nShortPath + asEdgeWeights[k];
			asPriority[count] = nShortPath + asEdgeWeights[k];
		}
		
	}
}
